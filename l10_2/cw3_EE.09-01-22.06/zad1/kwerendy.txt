FROM konta

FROM konta
INNER JOIN pytania ON konta_id = konta.id
WHERE pytania.id = 1;

FROM odpowiedzi
INNER JOIN konta
ON konta_id = konta.id
INNER JOIN pytania 
ON Pytania_id = pytania.id
WHERE pytania.id = 1
